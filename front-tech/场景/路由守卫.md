目的： 前端限制用户能访问到的页面
+ 未登录用户 —— 仅能访问白名单路由页面，其他 403
+ 登录用户 —— 都能访问（是否需要“登录回跳”）
配合 `react-router` 库一起实现

## `RouteGuard`
+ 白名单设计
+ 登录回跳设计

```jsx
import { Navigate, useLocation, Outlet } from "react-router"

const white_list = ["/login"];
const login_path = "/login";
const home_path = "/";

const RouteGuard = () => {
	const location = useLocation();
	const token = localStorage.getItem("access_token");
	const isLogin = Boolean(token);
	const isWhitePath = white_list.includes(location.pathname);
	
	if(!isLogin && !isWhitePath) {
		// 登录回跳 -- state 记录原本想去的路由页面
		return <Navigate to={login_path} replace state={{ form: location.pathname }} />
	}
	
	//	看产品设计 -- 登录后的用户允许在回登录页面吗？
	if(isLogin && location.pathname === login_path) {
		return <Navigate to={home_path} />
	}
	return <Outlet />
}

export defualt RouteGuard;
```

## `routes`
```js
import { createBrowserRouter } from "react-router";
import RouteGuard from "./RouteGuard";

const router = createBrowserRouter([
  {
    Component: RouteGuard, // 路由守卫
    children: [
      {
        path: "/login",
        lazy: async () => {
          const { default: App } = await import("@/App.jsx");
          return { Component: App };
        },
      },
      {
        path: "/",
        lazy: async () => {
          const { default: Index } = await import("@/pages/Index.jsx");
          return { Component: Index };
        },
      },
      {
        path: "/use-reducer",
        lazy: async () => {
          const { default: UseReducer } = await import("@/feature/use-reducer");
          return { Component: UseReducer };
        },
      },
    ],
  },
]);

export default router;
```
