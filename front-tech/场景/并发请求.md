const path =  
  "/postgres/clusters/c95847135a8041ff80c4e12f15b07c29/database_schemas";  
const request_address = "http://10.10.40.25:21080/api/v1" + path;  
const options = (i) => ({  
  request_address,  
  method: "POST",  
  body: {  
    characterSet: "UTF8",  
    lc_collate: "C",  
    lc_ctype: "C",  
    name: `test${i}`,  
  },  
  headers: {  
    "Content-Type": "application/json; charset=UTF-8",  
    Cookie: 'cookie'  
  },  
});  
const task = async (i) => {  
  const { request_address, method, body, headers } = options(i);  
  try {  
    const response = await fetch(request_address, {  
      method,  
      headers,  
      body: JSON.stringify(body),  
    });  
    if (response.ok) {  
      console.log(i + "请求成功");  
    } else {  
      const errorBody = await response.text();  
      console.log("请求失败", response.status, errorBody);  
    }  
  } catch (error) {  
    console.error("请求出错", error.message);  
  }  
};  
const runWithConcurrency = async (tasks, maxConcurrency) => {  
    const results = []  
    const executing = []  
    for(let i = 0; i < tasks.length; i++) {  
        const task = tasks[i]  
        // 到达最大并发数，等待其中一个执行完成  
        if(executing.length >= maxConcurrency) {  
            await Promise.race(executing)  
        }  
        // 启动一个任务  
        const promise = (async () => {  
            const res = await task()  
            executing.splice(executing.indexOf(promise), 1)  
            results.push(res)  
        })()  
        executing.push(promise)  
    }  
    // 等待其他任务完成  
    await Promise.all(executing)  
    return results  
}  
(async () => {  
    const tasks = []  
    for(let i = 6; i < 10; i++) {  
        tasks.push(() => task(i))  
    }  
    await runWithConcurrency(tasks, 3)  
})()
