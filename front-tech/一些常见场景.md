## 使用 antd Modal + Form

这里面主要注意的是，怎么通过 ref 来获取子组件中 Form 的数据

- 父组件 useRef
- 子组件 forwardRef + useImperativeHandle（暴露 dom 方法）+ useForm

其他的都是一些 antd 提供的 api，也需要注意一下

- form.validateFields() 获取验证后的表单数据
- Form.useWatch('xxx', form) 检测某个数据的变化
- form.getFieldsValue('xxx') 获取某个 name 的值
- form.setFieldsValue({xxx: 'xxx'}) 设置对应 name 的值

App 组件

```jsx
import { useRef, useState } from "react";
import { Modal, Button } from "antd";
import TestForm from "./TestForm";

const App = () => {
  const [isModalOpen, setIsModalOpen] = useState(false);
  const formRef = useRef(null);

  const showModal = () => {
    setIsModalOpen(true);
  };

  const handleOk = async () => {
    const res = await formRef.current.validateFields();
    console.log("res>>>", res);
    setIsModalOpen(false);
  };

  const handleCancel = () => {
    setIsModalOpen(false);
  };

  return (
    <>
      <Button type="primary" onClick={showModal}>
        登录/注册
      </Button>
      <Modal title="登录/注册" open={isModalOpen} onOk={handleOk} onCancel={handleCancel}>
        <TestForm ref={formRef} />
      </Modal>
    </>
  );
};
export default App;
```

子组件

```jsx
import React, { forwardRef, useEffect, useImperativeHandle } from "react";
import { Form, Input, Select, Radio } from "antd";
import useHobbies from "./hooks/useHobbies";

const TestForm = forwardRef((props, ref) => {
  const [form] = Form.useForm();
  const { hobbies } = useHobbies();
  const type = Form.useWatch("type", form);
  const isShow = type === "register";

  useImperativeHandle(ref, () => ({
    validateFields: () => form.validateFields(),
    getFieldsValue: (fieldName) => form.getFieldsValue(fieldName),
  }));

  const initialValues = useRef({
    type: "login",
    username: "新用户123",
    password: "123456",
    hobby: "",
  });

  useEffect(() => {
    if (hobbies.length > 0) {
      form.setFieldsValue({ hobby: hobbies[0]?.value });
    }
  }, [hobbies, form]);

  return (
    <>
      <Form
        form={form}
        initialValues={initialValues.current}
        labelCol={{ span: 4 }} // 标签占的位置
        wrapperCol={{ span: 15, offset: 2 }} // 右侧表单元素占的位置，向左偏移了2格
      >
        <Form.Item label={null} name="type" wrapperCol={{ offset: 0 }}>
          <Radio.Group
            block
            options={[
              { label: "登录", value: "login" },
              { label: "注册", value: "register" },
            ]}
            optionType="button"
            buttonStyle="solid"
            onChange={(e) => form.setFieldsValue({ type: e.target.value })}
          />
        </Form.Item>

        <Form.Item
          label="用户名"
          name="username"
          rules={[{ required: true, message: "请输入用户名" }]}
        >
          <Input />
        </Form.Item>

        <Form.Item label="密码" name="password" rules={[{ required: true, message: "请输入密码" }]}>
          <Input.Password />
        </Form.Item>

       <Form.Item
          noStyle
          shouldUpdate={(prevValues, curValues) => prevValues.type !== curValues.type}
        >
          {({ getFieldValue }) =>
            getFieldValue("type") === "register" ? (
              <Form.Item label="兴趣爱好" name="hobby">
                <Select options={hobbies} placeholder="请选择兴趣爱好" />
              </Form.Item>
            ) : null
          }
        </Form.Item>

        {/* {isShow && (
          <Form.Item label="兴趣爱好" name="hobby">
            <Select options={hobbies} placeholder="请选择兴趣爱好" />
          </Form.Item>
        )} */}
      </Form>
    </>
  );
});
export default TestForm;
```

自定义 hook 获取数据

```jsx
import { useEffect, useState } from "react";

const useHobbies = () => {
  const [hobbies, setHobbies] = useState([]);

  const getHobbies = async () => {
    try {
      const res = await fetch("http://localhost:3000/hobbies");
      const { data } = await res.json();
      const hobbies = data.map((v) => {
        return { label: v.hobby, value: v.msg };
      });
      setHobbies(hobbies);
    } catch (error) {
      console.log(error);
    }
  };

  useEffect(() => {
    getHobbies();
  }, []);

  return { hobbies };
};
export default useHobbies;
```
